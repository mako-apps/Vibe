interface CrossfadeOverlayProps {
    item: Message;
    isSequenceStart: boolean;
    isSequenceEnd: boolean;
    isMiddle: boolean;
    startX: number;
    startY: number;
    startWidth: number;
    startHeight: number;
    targetX: number;
    targetY: number;
    targetWidth: number;
    targetHeight: number;
    onComplete: () => void;
}

const CrossfadeOverlay = ({
    item,
    isSequenceStart,
    isSequenceEnd,
    isMiddle,
    startX, startY, startWidth, startHeight,
    targetX, targetY, targetWidth, targetHeight,
    onComplete,
}: CrossfadeOverlayProps) => {
    const progress = useSharedValue(0);

    useEffect(() => {
        progress.value = withTiming(1, {
            duration: 400,
            easing: Easing.out(Easing.cubic),
        }, (finished) => {
            if (finished) {
                runOnJS(onComplete)();
            }
        });
    }, []);

    const containerStyle = useAnimatedStyle(() => {
        const p = progress.value;

        const x = interpolate(p, [0, 1], [startX, targetX]);
        const y = interpolate(p, [0, 1], [startY, targetY]);
        const w = interpolate(p, [0, 1], [startWidth, targetWidth]);
        const h = interpolate(p, [0, 1], [startHeight, targetHeight]);
        const opacityY = interpolate(p, [0, 0.2], [10, 0], 'clamp');

        return {
            position: 'absolute',
            left: x,
            top: y,
            width: w,
            height: h,
            zIndex: 9999,
            transform: [{ translateY: opacityY }],
            opacity: interpolate(p, [0.2, 0.4], [0, 1], 'clamp'),
        };
    });

    return (
        <MessageBubble
            item={item}
            isMe={true}
            isSequenceStart={isSequenceStart}
            isSequenceEnd={isSequenceEnd}
            isMiddle={isMiddle}
            displayContent={item.plaintext || ''}
            style={containerStyle}
            shouldEnterFromInput={false}
            isClone={true}
        />
    );
};

const withAlpha = (color: string, alpha: number) => {
    if (!color) return `rgba(255, 255, 255, ${alpha})`
    if (color.startsWith('#')) {
        const r = parseInt(color.slice(1, 3), 16)
        const g = parseInt(color.slice(3, 5), 16)
        const b = parseInt(color.slice(5, 7), 16)
        return `rgba(${r}, ${g}, ${b}, ${alpha})`
    }
    return color
}

// Global decryption queue to prevent UI blocking
let decryptionQueue: Array<() => Promise<void>> = [];
let isProcessingQueue = false;

const processDecryptionQueue = async () => {
    if (isProcessingQueue || decryptionQueue.length === 0) return;
    isProcessingQueue = true;

    while (decryptionQueue.length > 0) {
        const task = decryptionQueue.shift();
        if (task) {
            await task();
            // Yield to UI between each decryption
            await new Promise(resolve => requestAnimationFrame(resolve));
        }
    }
    isProcessingQueue = false;
};

                withTiming(1, { duration: 1500 }),

            withTiming(50, { duration: 1500 }), // Move across

            false
        );
    }, []);

    const maskStyle = useAnimatedStyle(() => ({
        transform: [{ translateX: translateX.value }]
    }));

    return (
        <View style={{ height: 20, width: 80, overflow: 'hidden', alignItems: 'center' }}>
            <MaskedView
                style={{ height: '100%', width: '100%' }}

// Swipe threshold for triggering reply - reduced for better UX
const SWIPE_REPLY_THRESHOLD = 40;

const MessageItem = React.memo(({ item, isMe, onReply, onLongPress, chatId, shouldEnter = false, isSequenceStart, isSequenceEnd, isMiddle, showDateHeader, dateHeaderText, isHidden = false, replyPreview, senderName, onViewOnce, onSetItemRef, onItemLayout }: MessageItemProps) => {
    const { colors, effectiveTheme } = useThemeStore();
    const { updateMessageDecryption } = useChatStore();
    const { user } = useAuthStore();

    // Swipe animation values
    const swipeX = useSharedValue(0);
    const hasTriggeredHaptic = useSharedValue(false);

    // Use store plaintext directly - no local state duplication
    const displayText = item.plaintext;
    const needsDecryption = !displayText && !!item.encryptedContent;
    const [isDecrypting, setIsDecrypting] = useState(needsDecryption);
    const [decryptAttempted, setDecryptAttempted] = useState(false);

    // Haptic trigger function
    const triggerHaptic = useCallback(() => {
        Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);
    }, []);

    // Trigger reply callback
    const triggerReply = useCallback(() => {
        Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Light);
        onReply(item);
    }, [onReply, item]);

    // Pan gesture for swipe-to-reply
    const swipeGesture = Gesture.Pan()
        .activeOffsetX([-15, 15])
        .failOffsetY([-10, 10])
        .onUpdate((e) => {
            // Only allow swipe in one direction based on message owner
            const translation = isMe ? Math.min(0, e.translationX) : Math.max(0, e.translationX);
            const dampedTranslation = isMe
                ? Math.max(-SWIPE_REPLY_THRESHOLD * 1.5, translation)
                : Math.min(SWIPE_REPLY_THRESHOLD * 1.5, translation);

            swipeX.value = dampedTranslation;

            // Trigger haptic when threshold is crossed
            const absTranslation = Math.abs(dampedTranslation);
            if (absTranslation >= SWIPE_REPLY_THRESHOLD && !hasTriggeredHaptic.value) {
                hasTriggeredHaptic.value = true;
                runOnJS(triggerHaptic)();
            } else if (absTranslation < SWIPE_REPLY_THRESHOLD && hasTriggeredHaptic.value) {
                hasTriggeredHaptic.value = false;
            }
        })
        .onEnd((e) => {
            const absTranslation = Math.abs(swipeX.value);
            if (absTranslation >= SWIPE_REPLY_THRESHOLD) {
                runOnJS(triggerReply)();
            }
            // Use smooth timing instead of spring
            swipeX.value = withTiming(0, { duration: 200, easing: Easing.bezier(0.25, 0.1, 0.25, 1) });
            hasTriggeredHaptic.value = false;
        });

    // Animated style for swipe
    const swipeStyle = useAnimatedStyle(() => ({
        transform: [{ translateX: swipeX.value }]
    }));

    // Reply icon style - appears during swipe
    const replyIconStyle = useAnimatedStyle(() => {
        const absTranslation = Math.abs(swipeX.value);
        const progress = Math.min(absTranslation / SWIPE_REPLY_THRESHOLD, 1);
        const isThresholdReached = absTranslation >= SWIPE_REPLY_THRESHOLD;

        return {
            opacity: interpolate(progress, [0, 0.3, 1], [0, 0.5, 1]),
            transform: [
                { scale: interpolate(progress, [0, 1], [0.5, isThresholdReached ? 1.2 : 1]) },
                { translateX: isMe ? 10 : -10 }
            ]
        };
    });

    useEffect(() => {
        // Skip if already decrypted or no encrypted content
        if (displayText || !item.encryptedContent) {
            setIsDecrypting(false);
            return;
        }

        // Check if session is available
        const sess = AuthManager.getInstance().getSession();
        if (!sess?.keyPair?.privateKey) {

            return;
        }

        // Skip if already attempted and failed
        if (decryptAttempted && !displayText) {
            return;
        }

        let cancelled = false;
        setIsDecrypting(true);

        const decryptTask = async () => {
            if (cancelled) return;
            try {
                const text = await decryptMessage(sess.keyPair.privateKey, item.encryptedContent, item.fromId === sess.userId);
                if (!cancelled && chatId) {
                    updateMessageDecryption(chatId, item.id, text);
                }
            } catch (e) {
                console.warn('Decrypt item failed', e);
                if (!cancelled && chatId) {
                    updateMessageDecryption(chatId, item.id, "[Decryption Error]");
                }
            } finally {
                if (!cancelled) {
                    setIsDecrypting(false);
                    setDecryptAttempted(true);
                }
            }
        };

        const handle = InteractionManager.runAfterInteractions(() => {
            decryptionQueue.push(decryptTask);
            processDecryptionQueue();
        });

        return () => {
            cancelled = true;
            handle.cancel();
        };
    }, [item.encryptedContent, item.id, chatId, displayText, user?.userId]);

    // Determine what to display
    let displayContent: string;
    if (displayText) {
        displayContent = displayText;
    } else if (isDecrypting) {
        displayContent = "···";
    } else if (item.type === 'image') {
        displayContent = "📷 Image";
    } else {
        displayContent = "···";
    }

    const handleBubbleLongPress = (ref: any) => {
        ref.current?.measureInWindow((x: number, y: number, width: number, height: number) => {
            onLongPress(item, { x, y, width, height }, { displayContent, isSequenceStart, isSequenceEnd, isMiddle });
        });
    };

    return (
        <View>
            {showDateHeader && (
                <View style={styles.dateHeaderContainer}>
                    <SafeLiquidGlass style={styles.dateHeaderGlass} blurIntensity={15} tint={effectiveTheme === 'dark' ? 'dark' : 'light'}>
                        <Text style={[styles.dateHeaderText, { color: colors.text }]}>{dateHeaderText}</Text>
                    </SafeLiquidGlass>
                </View>
            )}
            {/* Sender name for group/channel messages */}
            {senderName && isSequenceStart && !isMe && (
                <Text style={{ color: colors.primary, fontSize: 12, fontWeight: '600', marginLeft: 14, marginBottom: 2 }}>
                    {senderName}
                </Text>
            )}
            <View style={[styles.swipeContainer, isMe ? styles.swipeContainerMe : styles.swipeContainerThem]}>
                {/* Reply Icon - positioned based on message owner, no background */}
                <Animated.View
                    style={[
                        styles.replyIconContainer,
                        isMe ? styles.replyIconRight : styles.replyIconLeft,
                        replyIconStyle
                    ]}
                >
                    <Reply size={20} color={colors.primary} />
                </Animated.View>

                <GestureDetector gesture={swipeGesture}>
                    <Animated.View
                        style={[
                            styles.msgRow,
                            isMe ? styles.msgRowMe : styles.msgRowThem,
                            swipeStyle
                        ]}
                    >
                        <View
                            style={{
                                alignSelf: isMe ? 'flex-end' : 'flex-start',
                                marginVertical: 1,
                                maxWidth: '85%',
                                marginHorizontal: 8,
                            }}
                        >
                            <View
                                ref={(ref: View | null) => onSetItemRef?.(item.id, ref)}
                                onLayout={() => onItemLayout?.(item.id)}
                                collapsable={false}
                            >
                                <MessageBubble
                                    item={item}
                                    isMe={isMe}
                                    displayContent={displayContent}
                                    isSequenceStart={isSequenceStart}
                                    isSequenceEnd={isSequenceEnd}
                                    isMiddle={isMiddle}
                                    isHidden={isHidden}
                                    onLongPress={handleBubbleLongPress}
                                    replyPreview={replyPreview}
                                    onViewOnce={onViewOnce}
                                    shouldEnterFromInput={shouldEnter && isMe}
                                    isClone={true}
                                />
                            </View>
                        </View>
                    </Animated.View>
                </GestureDetector>
            </View>
        </View>
    );
});

            const sess = AuthManager.getInstance().getSession();
            if (!sess?.keyPair?.privateKey) {

            const priorityMessages = sortedUndecrypted.slice(0, VISIBLE_COUNT);
            const remainingMessages = sortedUndecrypted.slice(VISIBLE_COUNT);

            // Phase 1: Decrypt visible messages with yields between each
            const priorityMap: Record<string, string> = {};
            for (let i = 0; i < priorityMessages.length; i++) {
                if (cancelled) break;
                const msg = priorityMessages[i];
                try {
                    const text = await decryptMessage(sess.keyPair.privateKey, msg.encryptedContent, msg.fromId === sess.userId);
                    priorityMap[msg.id] = text;
                } catch (e) {
                    priorityMap[msg.id] = "[Decryption Error]";

                // Yield after each decryption to prevent UI blocking
                if (i < priorityMessages.length - 1) await yieldToUI();

            // Update store and show chat immediately
            if (!cancelled && Object.keys(priorityMap).length > 0) {
                batchUpdateDecryption(activeChatId, priorityMap);
            }

            if (!cancelled) {
                initialDecryptDone.current = true;
                messagesLengthRef.current = chat.messages.length;

                        if (cancelled) break;

                            const text = await decryptMessage(sess.keyPair.privateKey, msg.encryptedContent, msg.fromId === sess.userId);
                            backgroundMap[msg.id] = text;
                            count++;

                            if (count % 3 === 0) {
                                batchUpdateDecryption(activeChatId, { ...backgroundMap });
                                Object.keys(backgroundMap).forEach(k => delete backgroundMap[k]);
                                // Longer yield for background work
                                await new Promise(r => setTimeout(r, 16)); // ~1 frame
                            }
                        } catch (e) {
                            backgroundMap[msg.id] = "[Decryption Error]";
                        }
                    }

                    // Final batch
                    if (!cancelled && Object.keys(backgroundMap).length > 0) {
                        batchUpdateDecryption(activeChatId, backgroundMap);
                    }
                }, 100); // Start background decryption after 100ms
            }

    const shimmerStyle = useAnimatedStyle(() => ({
        transform: [{ translateX: shimmerX.value }]
    }));

    const handleReply = useCallback((msg: Message) => {
        setReplyTo({
            messageId: msg.id,
            userName: msg.fromId === user?.userId ? 'You' : (chat?.friendName || 'User'),
            userHandle: msg.fromId === user?.userId ? user?.username : chat?.friendName?.toLowerCase().replace(/\s+/g, ''),
            content: msg.plaintext || ''
        });
    }, [user?.userId, user?.username, chat?.friendName]);

    const handleViewOnce = useCallback((messageId: string) => {
        useChatStore.getState().markViewOnceViewed(activeChatId || '', messageId);
    }, [activeChatId]);

    const handleSetItemRef = useCallback((id: string, ref: View | null) => {
        if (ref) itemRefs.current.set(id, ref);
        else itemRefs.current.delete(id);
    }, []);

    const handleItemLayout = useCallback((id: string) => {
        if (hiddenMessageIds.has(id)) {
            measureAndLaunchGhost(id);
        }
    }, [hiddenMessageIds, measureAndLaunchGhost]);

    const allUnreadCount = useMemo(() => {
        return chats.reduce((acc, c) => {
            if (c.chatId === activeChatId) return acc; // Don't count current chat
            return acc + (c.unreadCount || 0);
        }, 0);
    }, [chats, activeChatId]);

    return (
        <LinearGradient
            colors={wallpaperGradient as any}
            style={styles.container}
            pointerEvents="box-none"
        >
            <Stack.Screen options={{ headerShown: false }} />

            {/* WALLPAPER BACKGROUND - Moved to top level for immediate rendering */}
            <WallpaperBackground theme={resolvedTheme} />

            <GestureHandlerRootView style={{ flex: 1 }}>
                <GestureDetector gesture={chat ? gesture : Gesture.Native()}>
                    <Animated.View style={[{ flex: 1, flexDirection: 'row', width: SCREEN_WIDTH * 2 }, swipeStyle]}>

                        {/* --- PAGE 0: CHAT SCREEN --- */}
                        <View
                            style={{ width: SCREEN_WIDTH, height: '100%', position: 'relative' }}
                        >


                            {!activeChatId && !draftFriendId ? (
                                <View style={[styles.centerContainer, StyleSheet.absoluteFill]} pointerEvents="box-none">
                                    <EmptyChatState />
                                </View>
                            ) : !chat && !draftFriendId ? (
                                <View style={[styles.centerContainer, StyleSheet.absoluteFill]} pointerEvents="box-none">
                                    {isLoading ? (
                                        <Text style={{ color: colors.textSecondary }}>Loading chat...</Text>
                                    ) : (
                                        <View style={{ alignItems: 'center', gap: 10 }}>
                                            <Text style={[styles.loadingText, { color: colors.text }]}>Chat not found</Text>
                                            <TouchableOpacity onPress={() => loadChats()} style={{ padding: 10, backgroundColor: colors.elevated, borderRadius: 8 }}>
                                                <Text style={{ color: colors.text }}>Retry Sync</Text>
                                            </TouchableOpacity>
                                        </View>
                                    )}
                                </View>
                            ) : (
                                <>
                                    {/* HEADER MASK */}
                                    <View style={[styles.headerContainer, { height: insets.top + 80, zIndex: 60, pointerEvents: 'none' }]}>
                                        <MaskedViewAny
                                            style={StyleSheet.absoluteFill}
                                            maskElement={<LinearGradient colors={['rgba(0,0,0,1)', 'rgba(0,0,0,0)']} locations={[0.5, 1]} style={StyleSheet.absoluteFill} />}
                                        >
                                            <BlurView intensity={20} tint={effectiveTheme === 'dark' ? 'dark' : 'light'} style={[StyleSheet.absoluteFill, { backgroundColor: withAlpha(wallpaperGradient[0], 0.85) }]} />
                                        </MaskedViewAny>
                                    </View>

                                    {/* HEADER CONTENT */}
                                    <View style={[
                                        styles.headerContentContainer,
                                        {
                                            height: insets.top + 60,
                                            paddingTop: insets.top,
                                            zIndex: 130,
                                            flexDirection: 'row',
                                            alignItems: 'center',
                                            justifyContent: 'center',
                                            paddingHorizontal: 14
                                        }
                                    ]} pointerEvents="box-none">

                                        {/* Left: Back Button (Absolute) */}
                                        <View style={{ position: 'absolute', left: 14, top: insets.top + 8, zIndex: 150 }}>
                                            <SafeLiquidGlass style={{ borderRadius: 22, overflow: 'hidden' }} blurIntensity={20} tint={effectiveTheme === 'dark' ? 'dark' : 'light'}>
                                                <TouchableOpacity
                                                    onPress={handleBack}
                                                    style={{
                                                        minWidth: 44,
                                                        height: 44,
                                                        justifyContent: 'center',
                                                        alignItems: 'center',
                                                        flexDirection: 'row',
                                                        paddingHorizontal: allUnreadCount > 0 ? 12 : 0,
                                                        gap: 2,
                                                    }}
                                                >
                                                    <ChevronLeft color={colors.text} size={32} />
                                                    {allUnreadCount > 0 && (
                                                        <View style={{
                                                            backgroundColor: colors.text,
                                                            borderRadius: 10,
                                                            minWidth: 20,
                                                            height: 20,
                                                            justifyContent: 'center',
                                                            alignItems: 'center',
                                                            paddingHorizontal: 4,
                                                        }}>
                                                            <Text style={{ color: colors.background, fontSize: 11, fontWeight: 'bold' }}>
                                                                {allUnreadCount > 99 ? '99+' : allUnreadCount}
                                                            </Text>
                                                        </View>
                                                    )}
                                                </TouchableOpacity>
                                            </SafeLiquidGlass>
                                        </View>

                                        {/* Center: Data (Name + Status) */}
                                        <SafeLiquidGlass
                                            style={{ maxWidth: '60%', minWidth: 160, height: 44, borderRadius: 22, overflow: 'hidden' }}
                                            blurIntensity={20}
                                            tint={effectiveTheme === 'dark' ? 'dark' : 'light'}
                                        >
                                            <TouchableOpacity
                                                onPress={navigateToProfile}
                                                style={{ flex: 1, alignItems: 'center', justifyContent: 'center', paddingHorizontal: 12 }}
                                            >
                                                <Text style={[styles.headerTitle, { color: colors.text }]} numberOfLines={1}>
                                                    {displayName}
                                                </Text>
                                                {!!activityText ? (
                                                    <Text style={[styles.headerStatus, { color: colors.primary }]}>
                                                        {activityText}
                                                    </Text>
                                                ) : (
                                                    <Text style={[styles.headerStatus, { color: isOnline ? '#34d399' : colors.textSecondary }]}>
                                                        {isOnline ? 'Online' : 'Last seen recently'}
                                                    </Text>
                                                )}
                                            </TouchableOpacity>
                                        </SafeLiquidGlass>

                                        {/* Right: Avatar (Absolute) */}
                                        <View style={{ position: 'absolute', right: 14, top: insets.top + 8, zIndex: 150 }}>
                                            <SafeLiquidGlass
                                                style={{ height: 44, width: 44, borderRadius: 22, overflow: 'hidden', padding: 1.5 }}
                                                blurIntensity={20}
                                                tint={effectiveTheme === 'dark' ? 'dark' : 'light'}
                                            >
                                                <TouchableOpacity
                                                    onPress={navigateToProfile}
                                                    style={{ flex: 1, borderRadius: 21, overflow: 'hidden', backgroundColor: colors.elevated }}
                                                >
                                                    {(chat?.friendImage || draftFriendImage) ? (
                                                        <Image source={{ uri: (chat?.friendImage || draftFriendImage)! }} style={{ width: '100%', height: '100%' }} />
                                                    ) : (
                                                        <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center', backgroundColor: colors.primary }}>
                                                            <Text style={{ color: '#fff', fontSize: 18, fontWeight: '600' }}>
                                                                {displayName[0]?.toUpperCase() || '?'}
                                                            </Text>
                                                        </View>
                                                    )}
                                                </TouchableOpacity>
                                            </SafeLiquidGlass>
                                        </View>
                                    </View>

                                    <View style={{ flex: 1, position: 'relative' }}>
                                        <Animated.View style={[{ flex: 1 }, pageShiftStyle]}>
                                            {/* MESSAGES LIST - Always show immediately, decrypt in-place */}
                                            {messagesWithSequence.length === 0 ? (
                                                <EmptyChatState friendName={chat?.friendName || draftFriendName} />
                                            ) : (
                                                <Animated.FlatList
                                                    ref={flatListRef}
                                                    data={messagesWithSequence}
                                                    extraData={[messagesWithSequence.length, onlineUsers.size, typingUsers.size, contextMenuVisible, selectedMessageInfo]}
                                                    keyExtractor={item => item.id}
                                                    itemLayoutAnimation={LinearTransition.duration(200)}
                                                    renderItem={({ item, index }: { item: any; index: number }) => (
                                                        <MessageItem
                                                            key={item.id}
                                                            item={item}
                                                            index={index}
                                                            isMe={item.fromId === user?.userId}
                                                            shouldEnter={newMessageIds.has(item.id)}
                                                            isHidden={(contextMenuVisible && selectedMessageInfo?.item?.id === item.id) || hiddenMessageIds.has(item.id)}
                                                            onSetItemRef={handleSetItemRef}
                                                            onItemLayout={handleItemLayout}
                                                            onReply={handleReply}
                                                            onLongPress={handleMessageLongPress}
                                                            chatId={activeChatId || ''}
                                                            isSequenceStart={item.isSequenceStart}
                                                            isSequenceEnd={item.isSequenceEnd}
                                                            isMiddle={item.isMiddle}
                                                            showDateHeader={item.showDateHeader}
                                                            dateHeaderText={item.dateHeaderText}
                                                            replyPreview={item.replyPreview}
                                                            senderName={(isGroup || isChannel) && item.fromId !== user?.userId ? (chat?.friendName || item.fromId?.slice(0, 8)) : undefined}
                                                            onViewOnce={handleViewOnce}
                                                        />
                                                    )}
                                                    inverted
                                                    contentContainerStyle={[styles.listContent, { paddingBottom: footerHeight + 12, paddingTop: 25 }]}
                                                    showsVerticalScrollIndicator={false}
                                                    keyboardDismissMode="interactive"
                                                    keyboardShouldPersistTaps="handled"
                                                />
                                            )}

                                            {/* FOOTER MASK - Positioned at bottom of shifted area */}
                                            <View
                                                pointerEvents="none"
                                                style={{
                                                    position: 'absolute',
                                                    bottom: 0,
                                                    left: 0,
                                                    right: 0,
                                                    height: footerHeight + 30,
                                                    zIndex: 60,
                                                }}
                                            >
                                                <MaskedViewAny
                                                    style={StyleSheet.absoluteFill}
                                                    maskElement={
                                                        <LinearGradient
                                                            colors={['rgba(0,0,0,0)', 'rgba(0,0,0,1)']}
                                                            locations={[0, 0.5]}
                                                            style={StyleSheet.absoluteFill}
                                                        />
                                                    }
                                                >
                                                    <BlurView
                                                        style={[StyleSheet.absoluteFill, { backgroundColor: withAlpha(wallpaperGradient[wallpaperGradient.length - 1], 0.85) }]}
                                                        intensity={20}
                                                        tint={effectiveTheme === 'dark' ? 'dark' : 'light'}
                                                    />
                                                </MaskedViewAny>
                                            </View>

                                            {/* BOTTOM BAR (Input) - Sticky at the bottom of shifted area */}
                                            <Animated.View style={{ position: 'absolute', bottom: 0, left: 0, right: 0, zIndex: 65, overflow: 'visible' }}>
                                                <View style={{ overflow: 'visible' }}>
                                                    {/* Input Container */}
                                                    <View
                                                        ref={inputBarRef}
                                                        onLayout={updateInputBarMeasure}
                                                        style={{ paddingBottom: bottomInsetPadding, paddingTop: inputVerticalPadding, paddingHorizontal: 16, overflow: 'visible' }}
                                                    >
                                                        {canSend ? (
                                                            <ChatInput
                                                                text={text}
                                                                setText={setText}
                                                                onSend={handleSend}
                                                                onAttach={handleAttach}
                                                                onVibeQuery={handleVibeQuery}
                                                                onInputFocus={() => setGifPanelVisible(false)}
                                                                onGif={() => {
                                                                    if (gifPanelVisible) {
                                                                        setGifPanelVisible(false);
                                                                        return;
                                                                    }

                                                                    // Check if keyboard is open to prevent layout shift
                                                                    if (keyboardShift.value > 20) {
                                                                        isSwitchingToPanel.value = true;
                                                                    }

                                                                    Keyboard.dismiss();
                                                                    setGifPanelVisible(true);
                                                                }}
                                                                replyTo={replyTo}
                                                                onCancelReply={() => setReplyTo(null)}
                                                                manualClear={true}
                                                                onTypingStatusChange={(isTyping) => {
                                                                    if (!activeChatId) return;
                                                                    if (isTyping) useChatStore.getState().sendTyping(activeChatId);
                                                                    else useChatStore.getState().sendStopTyping(activeChatId);
                                                                }}
                                                            />
                                                        ) : (
                                                            <View style={{ alignItems: 'center', paddingVertical: 14 }}>
                                                                <Text style={{ color: colors.textSecondary, fontSize: 14 }}>
                                                                    {isChannel ? 'This is a broadcast channel' : 'You cannot send messages here'}
                                                                </Text>
                                                            </View>
                                                        )}
                                                    </View>
                                                </View>
                                            </Animated.View>
                                        </Animated.View>
                                    </View>

                                    <GifPanel
                                        visible={gifPanelVisible}
                                        progress={gifPanelProgress}
                                        onClose={() => setGifPanelVisible(false)}
                                        onSelectGif={(gif) => {
                                            handleGif(gif.url);
                                            setGifPanelVisible(false);
                                        }}
                                        height={gifPanelHeight}
                                    />


                                </>
                            )}
                        </View>

                        <View style={{ width: SCREEN_WIDTH, height: '100%' }}>
                            <UserProfile
                                visible={true}
                                onClose={navigateToChat}
                                username={displayName}
                                userId={chat?.friendId || draftFriendId || ''}
                                isOnline={isOnline}
                                profileImage={chat?.friendImage || draftFriendImage}
                                bio={friendProfile?.bio}
                                onBlock={() => {
                                    if (user?.userId && chat?.friendId) {
                                        apiClient.blockUser(user.userId, chat.friendId).then(() => {
                                            router.replace('/(tabs)/chats');
                                        }).catch(() => {
                                            alert('Failed to block user');
                                        });
                                    }
                                }}
                            />
                        </View>

                    </Animated.View>
                </GestureDetector>

                {contextMenuVisible && selectedMessageInfo?.item && (
                    <MessageContextMenu
                        visible={contextMenuVisible}
                        onClose={() => setContextMenuVisible(false)}
                        position={menuPosition}
                        isMe={selectedMessageInfo?.item?.fromId === user?.userId}
                        messageText={selectedMessageInfo?.item?.plaintext}
                        onReply={() => { console.log('Reply to', selectedMessageInfo?.item?.id); setContextMenuVisible(false); }}
                        onDelete={handleDeleteMessage}
                        onEdit={handleEditMessage}
                        onForward={handleForwardMessage}
                        onSelect={handleSelectMessage}
                        colors={colors}
                        displayContent={selectedMessageInfo?.displayContent}
                        item={selectedMessageInfo?.item}
                        isSequenceStart={selectedMessageInfo?.isSequenceStart}
                        isSequenceEnd={selectedMessageInfo?.isSequenceEnd}
                        isMiddle={selectedMessageInfo?.isMiddle}
                    />
                )}

                <ForwardMessageModal
                    visible={forwardModalVisible}
                    onClose={() => setForwardModalVisible(false)}
                    onSend={handleForwardSend}
                />

                {ghostData && (
                    <CrossfadeOverlay
                        item={ghostData.item}
                        isSequenceStart={ghostData.isSequenceStart}
                        isSequenceEnd={ghostData.isSequenceEnd}
                        isMiddle={ghostData.isMiddle}
                        startX={ghostData.startX}
                        startY={ghostData.startY}
                        startWidth={ghostData.startWidth}
                        startHeight={ghostData.startHeight}
                        targetX={ghostData.targetX}
                        targetY={ghostData.targetY}
                        targetWidth={ghostData.targetWidth}
                        targetHeight={ghostData.targetHeight}
                        onComplete={onGhostComplete}
                    />
                )}

            </GestureHandlerRootView>
        </LinearGradient>
    );
}
